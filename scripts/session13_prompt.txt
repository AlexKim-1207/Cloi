import requests, base64, json
URL = 'https://cloi.pages.dev/api/analyze'
CASES = [
  ('q001.jpg', '여성 셀카'),
  ('q003.jpg', '남성 outfit'),
  ('q010.jpg', '여성 룩북'),
  ('q020.jpg', '단일 가방'),
  ('q030.jpg', '캐주얼 outfit'),
]
for fname, label in CASES:
  with open(f'fashion-search/eval/queries/{fname}', 'rb') as f:
    b64 = base64.b64encode(f.read()).decode()
  r = requests.post(URL, json={'imageBase64': b64, 'mimeType': 'image/jpeg'}, timeout=60)
  j = r.json()
  print(f'\\n=== {fname} ({label}) ===')
  print(f'_source: {j.get(\"_source\")}')
  print(f'gender: {j.get(\"gender\")} price_tier: {j.get(\"price_tier\")}')
  for k, v in (j.get('categories') or {}).items():
    if v:
      print(f'  {k}: color={v.get(\"color\")} subtype={v.get(\"subtype\")} pattern={v.get(\"pattern\")} length={v.get(\"length\")}')
